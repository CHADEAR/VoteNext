bool jsonTruthy(JsonVariantConst v){
  if (v.isNull()) return false;
  if (v.is<bool>()) return v.as<bool>();
  if (v.is<int>()) return v.as<int>() != 0;
  if (v.is<long>()) return v.as<long>() != 0;
  if (v.is<const char*>()) {
    String s = trimLower(String(v.as<const char*>()));
    return (s == "true" || s == "1" || s == "yes");
  }
  return false;
}

// ================== RGB LED PWM ==================
void initRgbLed() {
#if USE_RGB_LED
  pinMode(PIN_LED_R, OUTPUT);
  pinMode(PIN_LED_G, OUTPUT);
  pinMode(PIN_LED_B, OUTPUT);

  ledcAttach(PIN_LED_R, RGB_PWM_FREQ, RGB_PWM_BITS);
  ledcAttach(PIN_LED_G, RGB_PWM_FREQ, RGB_PWM_BITS);
  ledcAttach(PIN_LED_B, RGB_PWM_FREQ, RGB_PWM_BITS);

  setRgbColor(0, 0, 0);
#endif
}

void setRgbColor(uint8_t red, uint8_t green, uint8_t blue) {
#if USE_RGB_LED
  uint32_t dutyR, dutyG, dutyB;

  if (RGB_COMMON_ANODE) {
    dutyR = RGB_PWM_MAX - red;
    dutyG = RGB_PWM_MAX - green;
    dutyB = RGB_PWM_MAX - blue;
  } else {
    dutyR = red;
    dutyG = green;
    dutyB = blue;
  }

  ledcWrite(PIN_LED_R, dutyR);
  ledcWrite(PIN_LED_G, dutyG);
  ledcWrite(PIN_LED_B, dutyB);
#else
  (void)red; (void)green; (void)blue;
#endif
}

void setRgbBoot() {
  setRgbColor(0, 0, RGB_BOOT_BLUE);
}

void updateStatusLed() {
#if USE_RGB_LED
  if (!screenOn) {
    setRgbColor(0, 0, 0);
    return;
  }

  if (systemBooting) {
    setRgbBoot();
    return;
  }

  bool wifiOk = (WiFi.status() == WL_CONNECTED);
  bool wsOk   = wsConnected;

  if (!wifiOk || !wsOk) {
    setRgbColor(RGB_LEVEL_RED, 0, 0);
  } else {
    setRgbColor(0, RGB_LEVEL_GREEN, 0);
  }
#endif
}

// ================== battery adc ==================
static void sortFloatArray(float *arr, int n) {
  for (int i = 0; i < n - 1; i++) {
    for (int j = i + 1; j < n; j++) {
      if (arr[j] < arr[i]) {
        float t = arr[i];
        arr[i] = arr[j];
        arr[j] = t;
      }
    }
  }
}

void initBatteryAdc() {
#if USE_BATTERY_ADC
  pinMode(PIN_BAT_ADC, INPUT);
  analogReadResolution(12);

  analogSetPinAttenuation(PIN_BAT_ADC, ADC_6db);

  for (int i = 0; i < 8; i++) {
    analogReadMilliVolts(PIN_BAT_ADC);
    delay(2);
  }
#endif
}

float readBatteryVoltage() {
#if USE_BATTERY_ADC
  float samples[BAT_ADC_SAMPLES];
  int count = 0;

  for (int i = 0; i < BAT_ADC_SAMPLES; i++) {
    uint32_t mv = analogReadMilliVolts(PIN_BAT_ADC);
    if (mv > 0) {
      float vPin = mv / 1000.0f;
      float vBat = vPin * ((BAT_R1 + BAT_R2) / BAT_R2);

      vBat = (vBat * BAT_CAL_FACTOR) + BAT_CAL_OFFSET;

      if (vBat >= BAT_MIN_VALID_V && vBat <= BAT_MAX_VALID_V) {
        samples[count++] = vBat;
      }
    }
    delay(2);
  }

  if (count <= 0) return 0.0f;

  sortFloatArray(samples, count);

  int trim = 0;
  if (count >= 10) trim = count / 5;

  float sum = 0.0f;
  int used = 0;

  for (int i = trim; i < count - trim; i++) {
    sum += samples[i];
    used++;
  }

  if (used <= 0) return 0.0f;
  return sum / used;
#else
  return 0.0f;
#endif
}

int batteryPercentFromVoltage(float v) {
  static const float lutV[] = {
    4.20, 4.15, 4.11, 4.08, 4.02,
    3.98, 3.95, 3.91, 3.87, 3.85,
    3.84, 3.82, 3.80, 3.79, 3.77,
    3.75, 3.73, 3.71, 3.69, 3.61, 3.50
  };

  static const int lutP[] = {
    100, 95, 90, 85, 80,
    75, 70, 65, 60, 55,
    50, 45, 40, 35, 30,
    25, 20, 15, 10, 5, 0
  };

  const int n = sizeof(lutV) / sizeof(lutV[0]);

  if (v >= lutV[0]) return 100;
  if (v <= lutV[n - 1]) return 0;

  for (int i = 0; i < n - 1; i++) {
    if (v <= lutV[i] && v >= lutV[i + 1]) {
      float vHigh = lutV[i];
      float vLow  = lutV[i + 1];
      int pHigh   = lutP[i];
      int pLow    = lutP[i + 1];

      float ratio = (v - vLow) / (vHigh - vLow);
      int p = (int)(pLow + (pHigh - pLow) * ratio + 0.5f);

      if (p < 0) p = 0;
      if (p > 100) p = 100;
      return p;
    }
  }

  return 0;
}

void updateBatteryStatus(bool force) {
#if USE_BATTERY_ADC
  if (!force && (millis() - lastBatteryReadMs < BAT_READ_INTERVAL_MS)) return;
  lastBatteryReadMs = millis();

  float v = readBatteryVoltage();

  if (v < BAT_MIN_VALID_V || v > BAT_MAX_VALID_V) {
    return;
  }

  if (!batteryReady || batteryVoltage <= 0.0f) {
    batteryVoltage = v;
  } else {
    batteryVoltage = batteryVoltage + (v - batteryVoltage) * BAT_FILTER_ALPHA;
  }

  float percentVoltage = batteryVoltage + BAT_PERCENT_COMP_V;
  if (percentVoltage > 4.20f) percentVoltage = 4.20f;
  if (percentVoltage < 3.50f) percentVoltage = 3.50f;

  int newPercent = batteryPercentFromVoltage(percentVoltage);

  if (!batteryReady || batteryPercent < 0) {
    batteryPercent = newPercent;
  } else {
    int diff = newPercent - batteryPercent;
    if (diff < 0) diff = -diff;

    if (diff >= BAT_PERCENT_HYST) {
      batteryPercent = newPercent;
    }
  }

  batteryReady = true;

#else
  (void)force;
#endif
}

String batteryText() {
#if USE_BATTERY_ADC
  if (!batteryReady || batteryPercent < 0) return "N/A";
  return String(batteryPercent) + "% (" + String(batteryVoltage, 2) + "V)";
#else
  return "OFF";
#endif
}

void tftSleep(bool sleepOn) {
#if USE_TFT_SLEEP_CMD
  if (sleepOn) {
    tft.writecommand(0x28); delay(10);
    tft.writecommand(0x10); delay(120);
  } else {
    tft.writecommand(0x11); delay(120);
    tft.writecommand(0x29); delay(10);
  }
#else
  (void)sleepOn;
#endif
}

void setScreenPower(bool on) {
  screenOn = on;
  voteModalOpen = false;
  voteConfirmMode = false;
  pendingVoteChoiceIdx = -1;

  if (!on) {
    ws.disconnect();
    wsConnected = false;
    tft.fillScreen(TFT_BLACK);
    setBacklightHW(false);
    tftSleep(true);
    updateStatusLed();
  } else {
    setBacklightHW(true);
    tftSleep(false);
    reconnectWs();
    statusDirty = true;
    goScreen((ui == SCR_SPLASH) ? SCR_STATUS : ui);
    updateStatusLed();
  }
}

// ================== text helpers ==================
static inline bool isUTF8Text(const String& s){
  for (size_t i = 0; i < s.length(); i++){
    if ((uint8_t)s[i] & 0x80) return true;
  }
  return false;
}

static inline void utf8PopOne(String& s){
  if(!s.length()) return;
  int i=(int)s.length()-1;
  while(i>0 && (((uint8_t)s[i]) & 0xC0)==0x80) i--;
  s.remove(i);
}

static inline void drawTextAuto(const String& s, int32_t x, int32_t y, uint8_t builtinFont, uint8_t datum){
  tft.setTextDatum(datum);
  if (thaiFontReady && isUTF8Text(s)) {
    tft.drawString(s, x, y);
  } else {
    tft.drawString(s, x, y, builtinFont);
  }
  tft.setTextDatum(TL_DATUM);
}

static String fitTextPxAuto(const String& s, int maxPx, uint8_t builtinFont) {
  if (maxPx <= 0) return "";

  int w = (thaiFontReady && isUTF8Text(s)) ? tft.textWidth(s) : tft.textWidth(s, builtinFont);
  if (w <= maxPx) return s;

  String out = s;
  while (out.length() > 0) {
    String trial = out + "...";
    int tw = (thaiFontReady && isUTF8Text(trial)) ? tft.textWidth(trial) : tft.textWidth(trial, builtinFont);
    if (tw <= maxPx) return trial;

    if (thaiFontReady && isUTF8Text(out)) utf8PopOne(out);
    else out.remove(out.length() - 1);
  }
  return "...";
}

void getVoteCardRectByVisibleIndex(int visibleIndex, int visibleCount,
                                   UIRect &card, UIRect &imgBox, UIRect &labelBox, UIRect &voteBtn){

  const int bottom = tft.height() - 16;
  const int leftPad = 16;
  const int rightPad = 22;
  const int gapX = 12;

  int areaH = bottom - top;

  int cols = visibleCount;
  if (cols < 1) cols = 1;
  if (cols > 3) cols = 3;

  int cardW = (areaW - (cols - 1) * gapX) / cols;

  if (cardW > 136) cardW = 136;
  if (cardH > 188) cardH = 188;
  if (cardH < 170) cardH = 170;

  int totalRowW = cols * cardW + (cols - 1) * gapX;
  int startX = leftPad + (areaW - totalRowW) / 2;
  int startY = top + (areaH - cardH) / 2;

  card = {
    (int16_t)(startX + visibleIndex * (cardW + gapX)),
    (int16_t)startY,

  };

static bool loadTouchCal(uint16_t cal[5], uint8_t rot){

  p.begin("touch", true);

  size_t len = p.getBytesLength("cal");

  p.end();
}

  p.end();
}

bool beginHttpClient(HTTPClient &http, WiFiClientSecure &secureClient, const String& url){
  secureClient.setInsecure();
  return http.begin(secureClient, url);
}

bool serverGet(const String& url, String& responseBody, int& statusCode, uint16_t timeoutMs = 8000){
  HTTPClient http;
  WiFiClientSecure secureClient;

  DBG("GET: " + url);

  http.setTimeout(timeoutMs);
  http.setReuse(false);
  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);

  if(!beginHttpClient(http, secureClient, url)){
    statusCode = -1;
    responseBody = "";
    DBG("GET begin failed");
    return false;
  }

  http.addHeader("Accept", "application/json");
  statusCode = http.GET();
  responseBody = http.getString();
  DBG("GET code: " + String(statusCode));
  http.end();
  return true;
}

bool serverPostJson(const String& url, const String& requestBody, String& responseBody, int& statusCode, uint16_t timeoutMs = 8000){
  HTTPClient http;
  WiFiClientSecure secureClient;

  DBG("POST: " + url);
  DBG("BODY: " + requestBody);

  http.setTimeout(timeoutMs);
  http.setReuse(false);
  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);

  if(!beginHttpClient(http, secureClient, url)){
    statusCode = -1;
    responseBody = "";
    return false;
  }

  http.addHeader("Content-Type", "application/json");
  http.addHeader("Accept", "application/json");

  statusCode = http.POST(requestBody);
  responseBody = http.getString();
  DBG("POST code: " + String(statusCode));
  http.end();
  return true;
}

// ================== JSON helpers ==================
static String pickStr(
  JsonVariantConst obj,
  const char* k1,
  const char* k2 = nullptr,
  const char* k3 = nullptr,
  const char* k4 = nullptr,
  const char* k5 = nullptr,
  const char* k6 = nullptr,
  const char* k7 = nullptr,
  const char* k8 = nullptr,
  const char* def = ""
){
  const char* keys[] = { k1, k2, k3, k4, k5, k6, k7, k8 };
  for(size_t i=0;i<8;i++){
    const char* k = keys[i];
    if(!k) continue;
    if(obj[k].is<const char*>()){
      const char* v = obj[k];
      if(v && *v) return String(v);
    }
  }
  return String(def);
}

}

// ================== image helpers ==================
String normalizeUrl(const String& u){
  if (!u.length()) return "";
  if (u.startsWith("http://") || u.startsWith("https://")) return u;
  if (u.startsWith("//")) return serverScheme() + ":" + u;
  if (u.startsWith("/"))  return String(SERVER_BASE_URL) + u;
  return String(SERVER_BASE_URL) + "/" + u;
}

String cloudinaryThumbUrl(const String& rawUrl, int w, int h){
  String url = normalizeUrl(rawUrl);
  if (!url.length()) return "";

  if (url.indexOf("res.cloudinary.com") < 0) return url;

  int pos = url.indexOf("/image/upload/");
  if (pos < 0) return url;

  String head = url.substring(0, pos + 14);
  String tail = url.substring(pos + 14);

  if (tail.startsWith("f_jpg") || tail.startsWith("c_fill") ||
      tail.startsWith("w_") || tail.startsWith("q_auto")) {
    return url;
  }

  String transform =
    "f_jpg,q_auto,c_fill,g_auto,w_" + String(w) + ",h_" + String(h) + "/";

  return head + transform + tail;
}

String imageCachePathForId(const String& id){
  String safe = id;
  safe.replace("/", "_");
  safe.replace("\\", "_");
  safe.replace(":", "_");
  safe.replace("?", "_");
  return "/c_v5_" + safe + ".jpg";
}

bool urlLooksLikeJpeg(String url){
  url.toLowerCase();
  int q = url.indexOf('?');
  if(q >= 0) url = url.substring(0, q);
  return url.endsWith(".jpg") || url.endsWith(".jpeg") || url.endsWith(".jfif");
}

bool httpDownloadToFile(const String& urlRaw, const String& path){
  if (!littlefsReady || WiFi.status() != WL_CONNECTED) return false;

  String url = cloudinaryThumbUrl(urlRaw, 72, 72);
  if (!url.length()) return false;

  HTTPClient http;
  WiFiClientSecure secureClient;

  http.setTimeout(15000);
  http.setReuse(false);
  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);

  if(!beginHttpClient(http, secureClient, url)) return false;

  int code = http.GET();
  if (code != 200) {
    http.end();
    return false;
  }

  String ctype = http.header("Content-Type");
  String ctypeL = ctype;
  ctypeL.toLowerCase();

  bool jpegOk =
    (ctypeL.indexOf("jpeg") >= 0) ||
    (ctypeL.indexOf("jpg") >= 0) ||
    (ctypeL.indexOf("jfif") >= 0) ||
    urlLooksLikeJpeg(url);

  if(!jpegOk){
    http.end();
    return false;
  }

  File f = LittleFS.open(path, "w");
  if (!f) {
    http.end();
    return false;
  }

  WiFiClient* stream = http.getStreamPtr();
  uint8_t buff[512];
  int remaining = http.getSize();
  uint32_t lastDataMs = millis();

  while (http.connected() && (remaining > 0 || remaining == -1)) {
    size_t avail = stream->available();
    if (avail) {
      int r = stream->readBytes(buff, (avail > sizeof(buff)) ? sizeof(buff) : avail);
      if (r > 0) {
        f.write(buff, r);
        if (remaining > 0) remaining -= r;
        lastDataMs = millis();
      }
    } else {
      delay(1);
      if(millis() - lastDataMs > 6000) break;
    }
  }

  f.close();
  http.end();

  return LittleFS.exists(path);
}

// ================== JPEG render ==================
static inline int16_t iMin(int16_t a, int16_t b){ return (a < b) ? a : b; }
static inline int16_t iMax(int16_t a, int16_t b){ return (a > b) ? a : b; }

static void renderJPEGClip(int16_t xpos, int16_t ypos,
                           int16_t clipX, int16_t clipY, int16_t clipW, int16_t clipH) {
  uint16_t jpgW = JpegDec.width;
  uint16_t jpgH = JpegDec.height;

  int16_t screenW = tft.width();
  int16_t screenH = tft.height();

  int16_t clipR = clipX + clipW;
  int16_t clipB = clipY + clipH;

  static uint16_t blockBuf[16 * 16];

  tft.startWrite();

  while (JpegDec.read()) {
    int16_t rawX = JpegDec.MCUx * JpegDec.MCUWidth  + xpos;
    int16_t rawY = JpegDec.MCUy * JpegDec.MCUHeight + ypos;

    int16_t rawW = JpegDec.MCUWidth;
    int16_t rawH = JpegDec.MCUHeight;

    if (rawX + rawW > xpos + (int16_t)jpgW) rawW = (xpos + (int16_t)jpgW) - rawX;
    if (rawY + rawH > ypos + (int16_t)jpgH) rawH = (ypos + (int16_t)jpgH) - rawY;

    if (rawW <= 0 || rawH <= 0) continue;

    int16_t visX0 = iMax(rawX, 0);
    int16_t visY0 = iMax(rawY, 0);
    int16_t visX1 = iMin((int16_t)(rawX + rawW), screenW);
    int16_t visY1 = iMin((int16_t)(rawY + rawH), screenH);

    visX0 = iMax(visX0, clipX);
    visY0 = iMax(visY0, clipY);
    visX1 = iMin(visX1, clipR);
    visY1 = iMin(visY1, clipB);

    int16_t visW = visX1 - visX0;
    int16_t visH = visY1 - visY0;

    if (visW <= 0 || visH <= 0) continue;

    uint16_t* src = (uint16_t*)JpegDec.pImage;
    int16_t srcStride = JpegDec.MCUWidth;

    int16_t offX = visX0 - rawX;
    int16_t offY = visY0 - rawY;

    bool fullBlockDirect =
      (rawW == JpegDec.MCUWidth) &&
      (rawH == JpegDec.MCUHeight) &&
      (offX == 0) &&
      (offY == 0) &&
      (visW == rawW) &&
      (visH == rawH);

    if (fullBlockDirect) {
      tft.pushImage(visX0, visY0, visW, visH, src);
    } else {
      for (int yy = 0; yy < visH; yy++) {
        memcpy(&blockBuf[yy * visW],
               &src[(offY + yy) * srcStride + offX],
               visW * sizeof(uint16_t));
      }
      tft.pushImage(visX0, visY0, visW, visH, blockBuf);
    }
  }

  tft.endWrite();
}

static bool drawJpegFromFSInBox(const String& path, int boxX, int boxY, int boxW, int boxH){
  if (!littlefsReady) return false;
  if (!LittleFS.exists(path)) return false;
  if (!JpegDec.decodeFsFile(path.c_str())) return false;

  int16_t drawX = boxX + (boxW - (int16_t)JpegDec.width)  / 2;
  int16_t drawY = boxY + (boxH - (int16_t)JpegDec.height) / 2;

  renderJPEGClip(drawX, drawY, boxX, boxY, boxW, boxH);
  return true;
}

      "imgUrl", "img", "avatar", "picture"
    );

  if (!isRemoteVoteMode(voteMode)) return;

  String rowShowId = pickStr(o, "show_id", "showId");
  if (rowShowId.length() && SHOW_ID.length() && !isPlaceholderShowId(SHOW_ID) && rowShowId != SHOW_ID) {
    return;
  }

  String status = pickStr(o, "status", "round_status", nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, "pending");
  if (!isDisplayableStatus(status)) return;

  String id = pickStr(o, "round_id", "roundId", "id", "uuid");
  if (!id.length()) return;

  for (int i=0;i<pollCount;i++){
    if (polls[i].id == id) return;
  }

  String title = pickStr(o, "title", "round_name", "name", "label", nullptr, nullptr, nullptr, nullptr, id.c_str());
  String desc  = pickStr(o, "description", "desc", "details", "note", "subtitle");
  String createdAt = pickStr(o, "created_at", "createdAt", "start_time");

  polls[pollCount++] = { id, title, desc, trimLower(status), createdAt, voteMode, rowShowId };
}

bool tryParsePollArray(JsonVariantConst src){
  if (!src.is<JsonArrayConst>()) return false;
  JsonArrayConst arr = src.as<JsonArrayConst>();
  for (JsonVariantConst v : arr) addPollFromObject(v);
  return true;
}

bool parsePollListFromJson(const String& body){
  DynamicJsonDocument doc(65536);
  if(deserializeJson(doc, body)) return false;

  pollCount = 0;

  if (tryParsePollArray(doc.as<JsonVariantConst>())) return (pollCount > 0);
  if (tryParsePollArray(doc["data"])) return (pollCount > 0);
  if (tryParsePollArray(doc["rooms"])) return (pollCount > 0);
  if (tryParsePollArray(doc["rounds"])) return (pollCount > 0);
  if (tryParsePollArray(doc["items"])) return (pollCount > 0);
  if (tryParsePollArray(doc["rows"])) return (pollCount > 0);

  if (doc["data"].is<JsonObjectConst>()) {
    JsonVariantConst data = doc["data"].as<JsonVariantConst>();
    if (tryParsePollArray(data["rooms"])) return (pollCount > 0);
    if (tryParsePollArray(data["rounds"])) return (pollCount > 0);
    if (tryParsePollArray(data["items"])) return (pollCount > 0);
    if (tryParsePollArray(data["rows"])) return (pollCount > 0);
  }

  JsonVariantConst one = doc["room"].is<JsonObjectConst>() ? doc["room"].as<JsonVariantConst>()
                        : doc["round"].is<JsonObjectConst>() ? doc["round"].as<JsonVariantConst>()
                        : doc["poll"].is<JsonObjectConst>() ? doc["poll"].as<JsonVariantConst>()
                        : doc["data"].is<JsonObjectConst>() ? doc["data"].as<JsonVariantConst>()
                        : doc.as<JsonVariantConst>();

  if (one.is<JsonObjectConst>()) {
    String id = pickStr(one, "round_id", "roundId", "id", "uuid");
    if (id.length()) {
      addPollFromObject(one);
      return (pollCount > 0);
    }
  }

  return false;
}

void buildFallbackPollFromActive(){
  if (pollCount > 0) return;
  if (!activeRoundId.length()) return;

  polls[0].id = activeRoundId;
  polls[0].title = activeRoundTitle.length() ? activeRoundTitle : "Active Poll";
  polls[0].description = "";
  polls[0].status = "voting";
  polls[0].createdAt = "";
  polls[0].voteMode = "remote";
  polls[0].showId = SHOW_ID;
  pollCount = 1;

  if (selectedPollIndex < 0) {
    selectedPollIndex = 0;
    selectedPollId = polls[0].id;
    selectedPollTitle = polls[0].title;
    selectedPollDesc = polls[0].description;
    selectedPollStatus = polls[0].status;
  }
}

bool fetchPollListHTTP(){
  if(WiFi.status() != WL_CONNECTED) return false;
  if(!SHOW_ID.length() || isPlaceholderShowId(SHOW_ID)) {
    DBG("SHOW_ID invalid");
    return false;
  }

  String body;
  bool ok = false;

  if (fetchRoomsPayload(body)) {
    DBG("Rooms body len: " + String(body.length()));
    ok = parsePollListFromJson(body);
  }

  if (!ok) {
    buildFallbackPollFromActive();
    ok = (pollCount > 0);
  }

  for (int i=0;i<pollCount-1;i++){
    for (int j=i+1;j<pollCount;j++){
      bool jVoting = isVotingStatus(polls[j].status);
      bool iVoting = isVotingStatus(polls[i].status);

      bool swapNeeded = false;
      if (jVoting && !iVoting) swapNeeded = true;
      else if (jVoting == iVoting && shortDate(polls[j].createdAt) > shortDate(polls[i].createdAt)) swapNeeded = true;

      if (swapNeeded) {
        PollItem tmp = polls[i];
        polls[i] = polls[j];
        polls[j] = tmp;
      }
    }
  }

  if (!activeRoundId.length()) {
    for (int i = 0; i < pollCount; i++) {
      if (isVotingStatus(polls[i].status)) {
        activeRoundId = polls[i].id;
        activeRoundTitle = polls[i].title;
        break;
      }
    }
  }

  if (selectedPollId.length()) {
    selectedPollIndex = -1;
    for (int i=0;i<pollCount;i++) {
      if (polls[i].id == selectedPollId) {
        selectedPollIndex = i;
        selectedPollStatus = polls[i].status;
        selectedPollTitle  = polls[i].title;
        selectedPollDesc   = polls[i].description;
        break;
      }
    }
  }

  if (selectedPollIndex < 0 && pollCount > 0) {
    selectedPollIndex = 0;
    selectedPollId    = polls[0].id;
    selectedPollTitle = polls[0].title;
    selectedPollDesc  = polls[0].description;
    selectedPollStatus= polls[0].status;
  }

  if (pollScroll >= pollCount) pollScroll = 0;

  DBG("pollCount = " + String(pollCount));
  return (pollCount > 0);
}

bool loadSelectedPollFromRoomsHTTP(const String& rid){
  if (WiFi.status() != WL_CONNECTED) return false;
  if (!rid.length()) return false;

  String body;
  if (!fetchRoomsPayload(body)) return false;

  DynamicJsonDocument doc(65536);
  if (deserializeJson(doc, body)) return false;

  JsonArrayConst arr;
  if (doc.is<JsonArrayConst>()) arr = doc.as<JsonArrayConst>();
  else if (doc["data"].is<JsonArrayConst>()) arr = doc["data"].as<JsonArrayConst>();
  else if (doc["rooms"].is<JsonArrayConst>()) arr = doc["rooms"].as<JsonArrayConst>();
  else if (doc["rounds"].is<JsonArrayConst>()) arr = doc["rounds"].as<JsonArrayConst>();
  else return false;

  for (JsonVariantConst v : arr) {
    if (!v.is<JsonObjectConst>()) continue;

    String id = pickStr(v, "round_id", "roundId", "id", "uuid");
    if (id != rid) continue;

    String voteMode = pickStr(v, "vote_mode", "voteMode");
    if (!isRemoteVoteMode(voteMode)) continue;

    String rowShowId = pickStr(v, "show_id", "showId");
    if (rowShowId.length() && SHOW_ID.length() && !isPlaceholderShowId(SHOW_ID) && rowShowId != SHOW_ID) {
      continue;
    }

    selectedPollId = rid;
    selectedPollTitle = pickStr(v, "title", "round_name", "name", "label");
    selectedPollDesc = pickStr(v, "description", "desc", "details", "note", "subtitle");
    selectedPollStatus = trimLower(pickStr(v, "status", "round_status", nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, "pending"));
    pollOpen = isVotingStatus(selectedPollStatus);

    JsonArrayConst choicesArr;
    if (v["contestants"].is<JsonArrayConst>())      choicesArr = v["contestants"].as<JsonArrayConst>();
    else if (v["choices"].is<JsonArrayConst>())     choicesArr = v["choices"].as<JsonArrayConst>();
    else if (v["options"].is<JsonArrayConst>())     choicesArr = v["options"].as<JsonArrayConst>();
    else if (v["items"].is<JsonArrayConst>())       choicesArr = v["items"].as<JsonArrayConst>();
    else if (v["candidates"].is<JsonArrayConst>())  choicesArr = v["candidates"].as<JsonArrayConst>();

    if (choicesArr.isNull()) {
      clearChoices();
      return false;
    }

    loadChoicesFromArray(choicesArr);
    return (choiceCount > 0);
  }

  return false;
}

bool fetchRoundDetailViaHttp(const String& rid){
  if(WiFi.status() != WL_CONNECTED) return false;
  if(!rid.length()) return false;

  if (loadSelectedPollFromRoomsHTTP(rid)) return true;

  String body;
  int code = 0;
  String url = String(SERVER_BASE_URL) + String(ROUND_DETAIL_HTTP_PATH) + rid;

  if(!serverGet(url, body, code, 8000)) return false;
  if(code != 200) return false;

  DynamicJsonDocument doc(65536);
  if(deserializeJson(doc, body)) return false;

  JsonVariantConst root;
  if(doc["data"].is<JsonObjectConst>() || doc["data"].is<JsonArrayConst>()) root = doc["data"].as<JsonVariantConst>();
  else root = doc.as<JsonVariantConst>();

  if(root["round"].is<JsonObjectConst>()) root = root["round"];
  if(!root.is<JsonObjectConst>()) return false;

  String voteMode = pickStr(root, "vote_mode", "voteMode");
  if (voteMode.length() && !isRemoteVoteMode(voteMode)) return false;

  String rowShowId = pickStr(root, "show_id", "showId");
  if (rowShowId.length() && SHOW_ID.length() && !isPlaceholderShowId(SHOW_ID) && rowShowId != SHOW_ID) {
    return false;
  }

  selectedPollId = rid;

  String ttl = pickStr(root, "title", "round_name", "name");
  if(ttl.length()) selectedPollTitle = ttl;

  String desc = pickStr(root, "description", "desc", "details", "note", "subtitle");
  selectedPollDesc = desc;

  String st = pickStr(root, "status", "round_status");
  if (st.length()) {
    selectedPollStatus = trimLower(st);
    pollOpen = isVotingStatus(selectedPollStatus);
  }

  JsonArrayConst arr;
  if(root["contestants"].is<JsonArrayConst>())            arr = root["contestants"].as<JsonArrayConst>();
  else if(root["choices"].is<JsonArrayConst>())           arr = root["choices"].as<JsonArrayConst>();
  else if(root["options"].is<JsonArrayConst>())           arr = root["options"].as<JsonArrayConst>();
  else if(root["items"].is<JsonArrayConst>())             arr = root["items"].as<JsonArrayConst>();
  else if(root["candidates"].is<JsonArrayConst>())        arr = root["candidates"].as<JsonArrayConst>();
  else if(root["round_contestants"].is<JsonArrayConst>()) arr = root["round_contestants"].as<JsonArrayConst>();
  else if(root["data"].is<JsonArrayConst>())              arr = root["data"].as<JsonArrayConst>();

  if(arr.isNull()){
    clearChoices();
    return false;
  }

  loadChoicesFromArray(arr);
  return (choiceCount > 0);
}

bool fetchActiveViaHttp(){
  if(WiFi.status() != WL_CONNECTED) return false;
  if(!SHOW_ID.length() || isPlaceholderShowId(SHOW_ID)) return false;

  auto tryUrl = [&](const String& url)->bool{
    int code = 0;
    String raw;
    if(!serverGet(url, raw, code, 6000)) return false;
    if(code != 200) return false;

    DynamicJsonDocument doc(49152);
    if(deserializeJson(doc, raw)) return false;

    JsonVariantConst root;
    if (doc["payload"].is<JsonObjectConst>() || doc["payload"].is<JsonArrayConst>()) root = doc["payload"].as<JsonVariantConst>();
    else if (doc["data"].is<JsonObjectConst>() || doc["data"].is<JsonArrayConst>()) root = doc["data"].as<JsonVariantConst>();
    else if (doc["active"].is<JsonObjectConst>() || doc["active"].is<JsonArrayConst>()) root = doc["active"].as<JsonVariantConst>();
    else root = doc.as<JsonVariantConst>();

    if (root.isNull()) {
      activeRoundId = "";
      activeRoundTitle = "";
      return true;
    }

    if (root["poll"].is<JsonObjectConst>()) root = root["poll"];
    if (root["round"].is<JsonObjectConst>()) root = root["round"];

    activeRoundId = pickStr(root, "id", "roundId", "round_id");
    activeRoundTitle = pickStr(root, "title", "round_name", "name");
    DBG("activeRoundId = " + activeRoundId);
    return true;
  };

  String u1 = String(SERVER_BASE_URL) + String(DEVICE_ACTIVE_HTTP_PATH1) + "?showId=" + SHOW_ID;
  if(tryUrl(u1)) return true;

  String u2 = String(SERVER_BASE_URL) + String(DEVICE_ACTIVE_HTTP_PATH2) + SHOW_ID;
  if(tryUrl(u2)) return true;

  return false;
}

}

void preloadChoiceImagesTick(){
  if (!needImagePreload) return;
  if (WiFi.status() != WL_CONNECTED) return;

  if (imgDlIndex >= choiceCount) {
    needImagePreload = false;
    voteDirty = true;
    return;
  }

  Choice &c = choices[imgDlIndex];
  if (c.imgUrl.length() && c.imgPath.length()) {
    if (LittleFS.exists(c.imgPath)) {
      c.imgReady = true;
    } else {
      bool ok = httpDownloadToFile(c.imgUrl, c.imgPath);
      c.imgReady = ok;
    }
  }
  imgDlIndex++;

  if (imgDlIndex >= choiceCount) {
    needImagePreload = false;
    voteDirty = true;
  }
}

// ================== vote ==================
void sendVote(int idx){
  if (!screenOn || idx < 0 || idx >= choiceCount) return;

  if (WiFi.status()!=WL_CONNECTED){
    toast("WiFi ยังไม่พร้อม");
    return;
  }

  if (!selectedPollId.length()){
    toast("ยังไม่ได้เลือก poll");
    return;
  }

  if (!isVotingStatus(selectedPollStatus)){
    toast("poll นี้ยังไม่เปิดโหวต");
    return;
  }

  String url = String(SERVER_BASE_URL) + String(DEVICE_VOTE_HTTP_PATH);

  StaticJsonDocument<256> doc;
  doc["deviceId"]     = deviceId();
  doc["showId"]       = SHOW_ID;
  doc["roundId"]      = selectedPollId;
  doc["contestantId"] = choices[idx].id;
  doc["value"]        = 1;

  String body;
  serializeJson(doc, body);

  int code = 0;
  String resp;
  if(!serverPostJson(url, body, resp, code, 8000)){
    toast("เชื่อมต่อเซิร์ฟเวอร์ไม่ได้");
    return;
  }

  if (code == 200) {
    lastVotedChoiceIdx = idx;
    showVoteModal();
  } else if (code == 409) {
    toast("โหวตซ้ำไม่ได้");
  } else if (code == 404) {
    toast("ไม่พบรอบโหวตนี้");
  } else {
    toast("โหวตไม่สำเร็จ: " + String(code));
  }
}

// ================== tap handler ==================
void handleTap(uint16_t x, uint16_t y){
  if(ui == SCR_STATUS){
    if(hitPad(BTN_JOIN, x, y, 12)){
      fetchActiveViaHttp();
      refreshPollList();

      if (pollCount <= 0) {
        toast("ไม่พบ poll remote");
      } else if (selectedPollIndex < 0) {
        selectPollByIndex(0);
      }

      goScreen(SCR_POLL);
      return;
    }
  }

  if(ui == SCR_POLL){
    if(hitPad(BTN_BACK, x, y, 12)){
      goScreen(SCR_STATUS);
      return;
    }

    int listTop = 66;
    int listBottom = BTN_JOIN_POLL.y - 12;
    int cardX = 22;
    int cardW = tft.width() - 48;
    int cardH = POLL_CARD_H;
    int gap = POLL_CARD_GAP;
    int availH = listBottom - listTop;
    int perPage = (availH + gap) / (cardH + gap);
    if (perPage < 1) perPage = 1;

    int end = pollScroll + perPage;
    if (end > pollCount) end = pollCount;

    for (int i = pollScroll; i < end; i++) {
      int idxOnScreen = i - pollScroll;
      int cardY = listTop + idxOnScreen * (cardH + gap);
      UIRect card = { (int16_t)cardX, (int16_t)cardY, (int16_t)cardW, (int16_t)cardH };

      if (hitPad(card, x, y, 8)) {
        selectPollByIndex(i);
        drawPoll();
        return;
      }
    }

    if(hitPad(BTN_JOIN_POLL, x, y, 12)){
      if(selectedPollIndex < 0 || !selectedPollId.length()){
        toast("ยังไม่ได้เลือก poll");
        return;
      }

      if(!isVotingStatus(selectedPollStatus)){
        toast("poll นี้ยังไม่เปิดโหวต");
        return;
      }

      if(choiceCount <= 0){
        bool ok = fetchRoundDetailViaHttp(selectedPollId);
        if(!ok){
          toast("ดึงข้อมูลช้อยส์ไม่ได้");
          return;
        }
      }

      voteScroll = 0;
      goScreen(SCR_VOTE);
      return;
    }
  }

  if(ui == SCR_VOTE){

    // ===== จัดการ modal ก่อน =====
    if (voteModalOpen) {
      if (voteConfirmMode) {
        if (hitPad(BTN_CONFIRM_BACK, x, y, 10)) {
          voteModalOpen = false;
          voteConfirmMode = false;
          pendingVoteChoiceIdx = -1;
          drawVote();
          return;
        }

        if (hitPad(BTN_CONFIRM_YES, x, y, 10)) {
          int idx = pendingVoteChoiceIdx;
          voteModalOpen = false;
          voteConfirmMode = false;
          pendingVoteChoiceIdx = -1;
          drawVote();
          sendVote(idx);
          return;
        }

        return;
      } else {
        if (hitPad(BTN_DONE, x, y, 10)) {
          voteModalOpen = false;
          goScreen(SCR_STATUS);
          return;
        }
        return;
      }
    }

    if(hitPad(BTN_BACK, x, y, 12)){
      goScreen(SCR_POLL);
      return;
    }

    int total = choiceCount;
    if (total <= 0) {
      toast("ไม่มีตัวเลือก");
      return;
    }

    if (voteScroll < 0) voteScroll = 0;
    int maxStart = ((total - 1) / VOTE_PER_PAGE) * VOTE_PER_PAGE;
    if (voteScroll > maxStart) voteScroll = maxStart;

    int visible = total - voteScroll;
    if (visible > VOTE_PER_PAGE) visible = VOTE_PER_PAGE;

    for(int cell=0; cell<visible; cell++){
      int idx = voteScroll + cell;
      UIRect card, imgBox, labelBox, vb;
      getVoteCardRectByVisibleIndex(cell, visible, card, imgBox, labelBox, vb);

      if(hitPad(vb, x, y, 12)){
        showVoteConfirmModal(idx);
        return;
      }
    }
  }
}

// ================== touch ==================
void handleTouch(){
  uint16_t x, y;
  bool down = readTouchOnce(x, y);
  uint32_t now = millis();

  if (down) {
    lastTouchX = x;
    lastTouchY = y;

    if (!touchDown) {
      if (now - lastTouchUpMs < TOUCH_DEBOUNCE_MS) return;

      touchDown = true;
      touchStartX = x;
      touchStartY = y;
      didDrag = false;
      touchStartMs = now;
      return;
    }

    int dy = (int)lastTouchY - (int)touchStartY;
    int dx = (int)lastTouchX - (int)touchStartX;

    if (!didDrag && (abs(dy) > TOUCH_DRAG_THRESHOLD || abs(dx) > TOUCH_DRAG_THRESHOLD)) {
      didDrag = true;

      if (ui == SCR_POLL && pollCount > 0) {
        int listTop = 66;
        int joinY = tft.height()-52;
        int listBottom = joinY - 12;
        int cardH = POLL_CARD_H, gap = POLL_CARD_GAP;
        int availH = listBottom - listTop;
        int perPage = (availH + gap) / (cardH + gap);
        if (perPage < 1) perPage = 1;

        if (dy < 0) {
          if (pollScroll + perPage < pollCount) pollScroll++;
        } else {
          if (pollScroll > 0) pollScroll--;
        }

        drawPoll();
        return;
      }

      int total = choiceCount;
      if (ui == SCR_VOTE && total > VOTE_PER_PAGE) {
        int maxStart = ((total - 1) / VOTE_PER_PAGE) * VOTE_PER_PAGE;

        if (dy < 0 || dx < 0) {
          voteScroll += VOTE_PER_PAGE;
          if (voteScroll > maxStart) voteScroll = maxStart;
        } else {
          voteScroll -= VOTE_PER_PAGE;
          if (voteScroll < 0) voteScroll = 0;
        }

        drawVote();
        return;
      }
    }
    return;
  }

  if (touchDown) {
    touchDown = false;
    lastTouchUpMs = now;

    if (!didDrag) {
      uint32_t heldMs = now - touchStartMs;
      if (heldMs >= TOUCH_TAP_MIN_MS && heldMs <= TOUCH_TAP_MAX_MS) {
        handleTap(lastTouchX, lastTouchY);
      }
    }

    didDrag = false;
  }
}

// ================== WS handler ==================
void wsEvent(WStype_t type, uint8_t* payload, size_t length) {
  if (!screenOn) return;

  switch (type) {
    case WStype_CONNECTED:
      wsConnected = true;
      requestActivePoll();
      statusDirty = true;
      updateStatusLed();
      break;

    case WStype_DISCONNECTED:
      wsConnected = false;
      statusDirty = true;
      updateStatusLed();
      break;

    case WStype_TEXT: {
      DynamicJsonDocument doc(49152);
      DeserializationError err = deserializeJson(doc, payload, length);
      if (err) {
        toast("WS JSON error");
        return;
      }

      String t = trimLower(String(doc["type"] | ""));
      JsonVariantConst p;

      if (doc["payload"].is<JsonObjectConst>() || doc["payload"].is<JsonArrayConst>()) p = doc["payload"].as<JsonVariantConst>();
      else if (doc["data"].is<JsonObjectConst>() || doc["data"].is<JsonArrayConst>()) p = doc["data"].as<JsonVariantConst>();
      else p = doc.as<JsonVariantConst>();

      if (!wsPayloadMatchesShow(p)) {
        return;
      }

      if (t == "active" || t == "poll_open") {
        String rid = extractRoundIdFromVar(p);
        activeRoundId = rid;

        JsonVariantConst src = p;
        if (src["poll"].is<JsonObjectConst>()) src = src["poll"];
        if (src["round"].is<JsonObjectConst>()) src = src["round"];
        activeRoundTitle = pickStr(src, "title", "round_name", "name");

        if (ui == SCR_POLL && pollCount <= 0 && activeRoundId.length()) {
          buildFallbackPollFromActive();
          pollDirty = true;
        }

        if (selectedPollId.length() && rid.length() && rid == selectedPollId) {
          if (ui != SCR_VOTE) {
            fetchRoundDetailViaHttp(selectedPollId);
            voteDirty = true;
          }
        }

        statusDirty = true;
      }

      if (t == "poll_close") {
        String rid = extractRoundIdFromVar(p);
        if (!rid.length() || rid == activeRoundId) {
          activeRoundId = "";
          activeRoundTitle = "";
        }

        if (selectedPollId.length() && rid == selectedPollId) {
          selectedPollStatus = "closed";
          pollOpen = false;
          voteDirty = true;
        }

        statusDirty = true;
      }
      break;
    }

    default: break;
  }
}

// ================== setup / loop ==================
void setup() {
#if SERIAL_DEBUG
  Serial.begin(115200);
  delay(200);
#else
  delay(50);
#endif

  initRgbLed();
  setRgbBoot();
  initBatteryAdc();
  updateBatteryStatus(true);

#if USE_PWR_BUTTON
  pinMode(PIN_PWR_BTN, INPUT_PULLUP);
#endif

#if USE_TFT_BL_PIN
  pinMode(PIN_TFT_BL, OUTPUT);
#endif

  pinMode(PIN_TOUCH_IRQ, INPUT);

  tft.init();
  tft.setRotation(TFT_ROTATION);
  tft.setSwapBytes(true);

  uiInitTheme();
  setBacklightHW(true);
  delay(30);

  littlefsReady = LittleFS.begin(true);
  if (littlefsReady && LittleFS.exists(THAI_FONT_FILE)) {
    tft.loadFont(THAI_FONT_NAME, LittleFS);
    thaiFontReady = true;
    DBG("[ThaiFont] LOADED");
  } else {
    thaiFontReady = false;
    DBG("[ThaiFont] NOT FOUND");
  }

  uint16_t cal[5] = { 0, 0, 0, 0, 0 };

#if FORCE_TOUCH_RECALIBRATE
  clearTouchCal();
#endif

  bool hasTouchCal = loadTouchCal(cal, TFT_ROTATION);

  if (!hasTouchCal) {
    tft.fillScreen(TFT_BLACK);
    tft.setTextColor(TFT_WHITE, TFT_BLACK);
    tft.setTextDatum(MC_DATUM);
    tft.drawString("Touch corners to calibrate", tft.width()/2, tft.height()/2 - 18, 2);
    tft.drawString("Tap all 4 corners", tft.width()/2, tft.height()/2 + 6, 2);
    tft.drawString("one by one", tft.width()/2, tft.height()/2 + 26, 2);
    tft.setTextDatum(TL_DATUM);
    delay(1200);
    setRgbBoot();

    tft.calibrateTouch(cal, TFT_MAGENTA, TFT_BLACK, 15);
    saveTouchCal(cal, TFT_ROTATION);

    tft.fillScreen(TFT_BLACK);
    tft.setTextColor(TFT_WHITE, TFT_BLACK);
    tft.setTextDatum(MC_DATUM);
    tft.drawString("Touch saved", tft.width()/2, tft.height()/2, 2);
    tft.setTextDatum(TL_DATUM);
    delay(700);
    setRgbBoot();
  }

  tft.setTouch(cal);

  resolveShowId();

  if (!SHOW_ID.length() || isPlaceholderShowId(SHOW_ID)) {
    DBG("SHOW_ID ยังไม่ถูกต้อง");
  } else {
    prefs.begin("vote_next", false);
    prefs.putString("showId", SHOW_ID);
    prefs.end();
  }

  goScreen(SCR_SPLASH);
  setRgbBoot();

  uint32_t bootLedStart = millis();
  while (millis() - bootLedStart < BOOT_COLOR_HOLD_MS) {
    setRgbBoot();
    delay(20);
  }

  startWiFi();

  uint32_t start = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - start < 12000) {
    delay(50);
    updateStatusLed();
  }

  updateBatteryStatus(true);

  if (WiFi.status() == WL_CONNECTED) {
    fetchActiveViaHttp();
    refreshPollList();

    if (pollCount > 0 && selectedPollIndex < 0) {
      selectPollByIndex(0);
    } else if (selectedPollId.length()) {
      fetchRoundDetailViaHttp(selectedPollId);
    }
  }

  reconnectWs();
  statusDirty = true;
  goScreen(SCR_STATUS);

  systemBooting = false;
  updateStatusLed();
}

void loop() {
  maintainWiFi();
  updateBatteryStatus(false);

  if (screenOn) {
    ws.loop();
    handleTouch();
    preloadChoiceImagesTick();

    if (statusDirty && ui == SCR_STATUS && millis() > toastHoldUntil) {
      statusDirty = false;
      drawStatus();
    }

    if (pollDirty && ui == SCR_POLL && millis() > toastHoldUntil) {
      pollDirty = false;
      drawPoll();
    }

    if (voteDirty && ui == SCR_VOTE && !voteModalOpen && millis() > toastHoldUntil) {
      voteDirty = false;
      drawVote();
    }

    if (WiFi.status() == WL_CONNECTED && millis() - lastRefreshMs > REFRESH_MS) {
      lastRefreshMs = millis();

      if (ui == SCR_STATUS || ui == SCR_POLL) {
        fetchActiveViaHttp();
      }

      if (ui == SCR_POLL || ui == SCR_STATUS) {
        refreshPollList();
      }

      if (ui == SCR_POLL) {
        pollDirty = true;
      }

      if (ui == SCR_STATUS) {
        statusDirty = true;
      }

      if (ui != SCR_VOTE && wsConnected) {
        requestActivePoll();
      }
    }
  }

  updateStatusLed();

#if USE_PWR_BUTTON
  static int lastBtn = HIGH;
  static uint32_t pressStartMs = 0;
  int now = digitalRead(PIN_PWR_BTN);

  if (lastBtn == HIGH && now == LOW) {
    pressStartMs = millis();
  }

  if (lastBtn == LOW && now == HIGH) {
    if (millis() - pressStartMs >= 800) {
      setScreenPower(!screenOn);
    }
  }

  lastBtn = now;
#endif
}
